    const match = md.match(/^---\n([\s\S]*?)\n---\n([\s\S]*)$/);
    if (!match) return { meta: {}, content: md };

    const meta = {};
    match[1].split('\n').forEach(line => {
        const [key, ...vals] = line.split(':');
